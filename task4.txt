2. Write a program to convert half of the string uppercase and rest half of string in lower case.
name = ("hello")

for i in range(len(string)):
    if i < len(string) // 2:
        name += string[i].upper()
    else:
        name += string[i].lower()

print("Original string:", string)
print("New string:", name)

3. count the number of any character occuring in the string without using count function.
name = input("Enter a name: ")
frute = input("Enter a character: ")
count = 0

for i in name:
    if i == frute:
        count += 1

print("The character '{}' appears {} times in the string.".format(frute, count))
